318 private int CalculateDistance(string scannedSource, string expectedSource)
319 {
320
321 var scannedSourceLength = scannedSource.Length;
322 var expectedSourceLength = expectedSource.Length;
323
324 var matrix = new int[scannedSourceLength + 1, expectedSourceLength + 1];
325
326 // Überprüft ob der erste string leer ist, wenn ja ist die Distanz die länge des zweiten strings.
327 if (scannedSourceLength == 0)
328 return expectedSourceLength;
329
330 // Überprüft ob der zweite string leer ist, wenn ja ist die Distanz die länge des ersten strings.
331 if (expectedSourceLength == 0)
332 return scannedSourceLength;
333
334
335 // Initialisierung der Zeilenanzahl des ersten string in der Matrix. Jedes char des stings, ist dabei eine Zeile.
336 for (var row = 0; row <= scannedSourceLength; row++)
337 {
338 matrix[row, 0] = row;
339 }
340 // Initialisierung der Spaltenanzahl des zweiten string in der Matrix. Jedes char des stings, ist dabei eine Sapalte.
341 for (var column = 0; column <= expectedSourceLength; column++)
342 {
343 matrix[0, column] = column;
344 }
345
346 // Hier wird die Distanz berechnet bzw. rausgefunden d.h. hier beginnt die eigetnliche Suche nach den Unterschieden.
347 // Dabei wird immer das Minimum innerhalb der temporären/ aktuellen matrix gesucht. Dies geschieht bis man an der Stelle n x m ankommt
348 for (var i = 1; i <= scannedSourceLength; i++)
349 {
350 for (var j = 1; j <= expectedSourceLength; j++)
351 {
352 int substitutionCount;
353 if (expectedSource[j - 1] == scannedSource[i - 1]) // Unterschied der zwei chars, wenn die gleich sind,
354 {
355 substitutionCount = 0; // dann ist die Differenz 0
356 }
357 else
358 {
359 substitutionCount = 1; // ansonsten 1.
360 }
361
362 matrix[i, j] = Math.Min(Math.Min(matrix[i - 1, j] + 1, // delete
363 matrix[i, j - 1] + 1), // insert
364 matrix[i - 1, j - 1] + substitutionCount); // replace
365
366 }
367 }
368 // Gibt die Distanz wieder zurück.
369 return matrix[scannedSourceLength, expectedSourceLength];
370 }
