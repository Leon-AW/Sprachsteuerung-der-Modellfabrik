149 public string FormResponse(string topic, string message, int commandPosition, string response)
150 {
151 if (topic.Equals("f/o/order")) //Wenn eine Bestellung gemacht wurde...

203 private void client_MqttPublishReceived(object sender, MqttMsgPublishEventArgs e)
204 {
205 var message = Encoding.UTF8.GetString(e.Message); //message wird gelesen
206 switch (e.Topic) //Wenn das Topic...

212 GetStockDataFromJson(message);
213 break;
214 case "f/i/order":
215 OrderState(message);
216 break;
217 default:
218 OrderPosition(message); //Wird ausgeführt wenn die anderen Fälle nicht eintreffen und state im Topping enthalten ist. (If Abfrage eigentlich nicht nötig, nur falls beim Subscriben erweitert werden sollte)
219 break;
220
221 } //kann bei Bedarf mit weiteren Befehlen erweitert werden

261 private void OrderPosition(string message)
262 {
263 try
264 {
265 dynamic json = JObject.Parse(message); //JsonObject
266 var active = (byte)json.active; //liest ob das Modul aktiv ist
267 var stationName = json.station.ToString(); //liest Stationennamen
268
269 if (_orderState.Equals("ORDERED")) //Wenn Bestellung bestellt ist
270 {
271 _workpieceLocation = " wird vom Lager zur Multi-Bearbeitungsstation befördert."; //dann Location
272 }
273
274 if (_orderState.Equals("IN_PROCESS")) //Wenn in Process ist (Multibearbeitungsstation)
275 {
276 if (active == 1) //dann --> wenn Modul aktiv ist
277 {
278 _workpieceLocation = GetWorkstationNameOf(stationName); //Stationname herausfinden
279 }
280 }
281 if (_orderState.Equals("SHIPPED") || _orderState.Equals("WAITING_FOR_ORDER")) //Wenn Bestellung ferig ist
282 {
283 _orderPosition = "Es ist keine Bestellung in Auftrag."; //dann keine Bestellung in Auftrag
284 }
285 else
286 {
287 _orderPosition = "Das " + _orderedWorkpieceColor + _workpieceLocation; //Ansonsten Antwort formen
288 }
289 }
290 catch (Exception e)
291 {
292 Console.WriteLine(@"Beim Lesen des MQTT Befehls (OrderPosition) ist ein Fehler aufgestreten" + e);
293 }
294 }
